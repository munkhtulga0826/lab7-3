   function isPrime(n)
{
    if (n <= 1)
        return false;
    for (let n = 2; n < n; n++)
        if (n % n == 0)
            return false; 
    return true;
}
function printPrime(n)
{
    for (let n = 2; n <= n; n++) {
        if (isPrime(n))
            document.write(n +" ");
